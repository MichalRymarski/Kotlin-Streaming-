x-data: Declares a new Alpine component scope.

<div x-data="{ count: 0 }">
  <button x-on:click="count++">Increment</button>
  <span x-text="count"></span>
</div>

x-init: Runs when the component is initialized.

<div x-data="{ message: 'Hello, world!' }" x-init="console.log(message)">
  <p x-text="message"></p>
</div>

x-show: Toggles the visibility of an element.

<div x-data="{ isVisible: false }">
  <button x-on:click="isVisible = !isVisible">Toggle Visibility</button>
  <p x-show="isVisible">This text will be shown or hidden based on the value of isVisible.</p>
</div>

x-bind: Dynamically sets HTML attributes.

<div x-data="{ isActive: true }">
  <div x-bind:class="{ 'active': isActive }">This div will have the 'active' class added when isActive is true.</div>
</div>

x-on: Attaches event listeners to elements.

<div x-data="{ count: 0 }">
  <button x-on:click="count++">Increment</button>
  <span x-text="count"></span>
</div>

x-model: Creates two-way data bindings.

<div x-data="{ name: 'John Doe' }">
  <input type="text" x-model="name" />
  <p>Hello, <span x-text="name"></span>!</p>
</div>

x-text: Updates the text content of an element.

<div x-data="{ message: 'Hello, world!' }">
  <p x-text="message"></p>
</div>

x-html: Updates the inner HTML of an element.

<div x-data="{ content: '<strong>This is bold text.</strong>' }">
  <div x-html="content"></div>
</div>

x-ref: References elements in your component.

<div x-data>
  <button x-ref="myButton">Click me</button>
  <script>
    document.addEventListener('alpine:init', () => {
      Alpine.data('component', () => ({
        init() {
          this.$refs.myButton.addEventListener('click', () => {
            alert('Button clicked!');
          });
        }
      }));
    });
  </script>
</div>

x-if: Conditionally renders elements.

<div x-data="{ showContent: false }">
  <button x-on:click="showContent = !showContent">Toggle Content</button>
  <template x-if="showContent">
    <div>This content will be shown or hidden based on the value of showContent.</div>
  </template>
</div>

x-for: Iterates over arrays or objects.

<div x-data="{ items: ['Apple', 'Banana', 'Cherry'] }">
  <ul>
    <template x-for="item in items">
      <li x-text="item"></li>
    </template>
  </ul>
</div>

x-transition: Applies transitions to elements.

<div x-data="{ isVisible: false }">
  <button x-on:click="isVisible = !isVisible">Toggle Visibility</button>
  <div x-show="isVisible" x-transition>
    This content will fade in and out when the button is clicked.
  </div>
</div>

x-cloak: Hides elements until Alpine is fully loaded.

<div x-data="{ message: 'Hello, world!' }" x-cloak>
  <p x-text="message"></p>
</div>

x-effect: Runs a callback when its dependencies change.

<div x-data="{ count: 0 }">
  <button x-on:click="count++">Increment</button>
  <div x-effect="console.log(`The count is ${count}`)"></div>
</div>

x-ignore: Ignores an element and its children for compilation.

<div x-data="{ message: 'Hello, world!' }">
  <div x-ignore>
    <p>This content will be ignored by Alpine.</p>
    <p x-text="message"></p>
  </div>
</div>

x-id: Generates unique IDs for elements.

<div x-data>
  <label :for="$id('my-input')">Enter your name:</label>
  <input type="text" :id="$id('my-input')" />
</div>

x-teleport: Moves an element to a different part of the DOM.

<div x-data="{ isModalOpen: false }">
  <button x-on:click="isModalOpen = true">Open Modal</button>
  <template x-teleport="body">
    <div x-show="isModalOpen" class="modal">
      <div class="modal-content">
        <span class="close-button" x-on:click="isModalOpen = false">×</span>
        <p>This modal will be teleported to the body of the document.</p>
      </div>
    </div>
  </template>
</div>

x-modelable: Creates custom two-way data binding.

<div x-data="{ value: 'Initial value' }">
  <input type="text" x-modelable="value" />
  <p>The value is: <span x-text="value"></span></p>
</div>

x-scroll: Adds scroll-based animations and interactions.

<div x-data="{ scroll: 0 }" x-scroll="scroll = $event.target.scrollTop">
  <p x-text="`Scroll position: ${scroll}`"></p>
</div>

x-scroll-snap: Enables snap scrolling.

<div x-data="{ snap: 'start' }" x-scroll-snap="snap">
  <p>Scroll here!</p>
</div>

x-scroll-class: Applies a class based on scroll position.

<div x-data="{ scrolled: false }" x-scroll="scrolled = ($event.target.scrollTop > 100)" x-scroll-class="scrolled: 'bg-white'">
  <p>Scroll down more than 100px to see the background change!</p>
</div>

x-scroll-progress: Tracks the scroll progress.

<div x-data="{ progress: 0 }" x-scroll-progress="progress">
  <p x-text="`Scroll progress: ${progress}`"></p>
</div>
